\section{Model Description}
\begin{equation}
I(string)=\begin{cases}
1 & \text{$string$ contains $key$},\\
0 & \text{$string$ contains no $key$}.
\end{cases}\\
\end{equation}
\begin{equation}
C(string)=\text{Number of word $key$ in $string$}\\
\end{equation}

\begin{equation}
score=\sum_{i=1}^{m} (nw*I(name)+tw*I(title)+cw*C(content)) / m 
\end{equation}
